Angular

Query your API without worrying about validations

Class PostsController {
    constructor(API){
      'ngInject';
      this.API = API;
      this.posts;
    }

    getPosts(){
        this.API.all('posts').get('')
        .then((response) => {
              this.posts = response.data;
        });
    }
}
